func nextPermutation(nums []int) {
	var i int
	for i = len(nums) - 2; i >= 0; i-- {
		if nums[i] < nums[i+1] {
			for j := len(nums) - 1; j >= i; j-- {
				if nums[i] < nums[j] {
					nums[i], nums[j] = nums[j], nums[i]
					break
				}
			}
			break
		}
	}

}

func reverse(a []int) {
	for i, n := 0, len(a); i < n/2; i++ {
		a[i], a[n-1-i] = a[n-1-i], a[i]
	}
}
