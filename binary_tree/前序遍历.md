func preorderTraversal(root *TreeNode) []int {
	var traversal func(node *TreeNode) // 声明一个函数
	res := make([]int, 0)
	traversal = func(node *TreeNode) { // 设置函数功能
		if node == nil {
			return
		}
		res = append(res, node.Val)
		traversal(node.Left)
		traversal(node.Right)
	}
	traversal(root) // 调用递归函数
	return res
}
